136static void TareaMedicion(void *pvParameter) {
137 while (true) {
138 // Esperar notificación del timer
139 ulTaskNotifyTake(pdTRUE, portMAX_DELAY); //recibe la notificación del timer para realizar la medición
140 if (MEDIR) {
141 // Realizar medición
142 distancia_actual = HcSr04ReadDistanceInCentimeters();
143
144 // Actualizar distancia mostrada si no está en HOLD
145 if (!HOLD) {
146 distancia_mostrada = distancia_actual;
147 }
148
149 // Mostrar distancia en LCD
150 LcdItsE0803Write(distancia_mostrada);
151
152 // Controlar LEDs según distancia actual
153 if (distancia_actual < 10) {
154 LedsOffAll();
155 } else if (distancia_actual <= 20) {
156 LedsOffAll();
157 LedOn(LED_1);
158 } else if (distancia_actual <= 30) {
159 LedsOffAll();
160 LedOn(LED_1);
161 LedOn(LED_2);
162 } else {
163 LedsOffAll();
164 LedOn(LED_1);
165 LedOn(LED_2);
166 LedOn(LED_3);
167 }
168 } else {
169 // Apagar LEDs y display cuando no se mide
170 LedsOffAll();
171 LcdItsE0803Off();
172 }
173 }
174}
