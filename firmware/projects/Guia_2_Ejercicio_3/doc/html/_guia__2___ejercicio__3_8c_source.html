170// Envía una señal (notificación) a la tarea de medición para decirle que es momento de realizar una nueva medicion
171
181static void TareaMedicion(void *pvParameter) {
182 while (true) {
183 // Esperar notificación del timer
184 ulTaskNotifyTake(pdTRUE, portMAX_DELAY); //recibe la notificación del timer para realizar la medición
185 if (MEDIR) {
186 // Realizar medición
187 distancia_actual = HcSr04ReadDistanceInCentimeters();
188
189 // Actualizar distancia mostrada si no está en HOLD
190 if (!HOLD) {
191 distancia_mostrada = distancia_actual;
192 }
193
194 // Mostrar distancia en LCD
195 LcdItsE0803Write(distancia_mostrada);
196
197 // Enviar distancia por UART
198 EnviarDistanciaUART(distancia_mostrada);
199
200 // Controlar LEDs según distancia actual
201 if (distancia_actual < 10) {
202 LedsOffAll();
203 } else if (distancia_actual <= 20) {
204 LedsOffAll();
205 LedOn(LED_1);
206 } else if (distancia_actual <= 30) {
207 LedsOffAll();
208 LedOn(LED_1);
209 LedOn(LED_2);
210 } else {
211 LedsOffAll();
212 LedOn(LED_1);
213 LedOn(LED_2);
214 LedOn(LED_3);
215 }
216 } else {
217 // Apagar LEDs y display cuando no se mide
218 LedsOffAll();
219 LcdItsE0803Off();
220 }
221 }
222}

236void app_main(void) {
237 // Inicializar sensor HC-SR04

242

246 // Inicializar switches
247 SwitchesInit();
248
249 // Configurar UART
250 serial_config_t uart_pc = {
251 .port = UART_PC,
252 .baud_rate = 115200,
253 .func_p = LeerPuertoSerie, /* Función que se ejecuta al recibir dato */
254 .param_p = NULL
255 };
256 UartInit(&uart_pc);
257
258 // Configurar interrupciones para switches
259 SwitchActivInt(SWITCH_1, InterrupcionTec1, NULL);
260 SwitchActivInt(SWITCH_2, InterrupcionTec2, NULL);
261
262 // Crear tarea de medición
263 xTaskCreate(&TareaMedicion, "MEDICION", 2048, NULL, 5, &tarea_medicion_handle);
264
265 // Configurar timer para mediciones cada 1 segundo
266 timer_config_t timer_medir = {
267 .timer = TIMER_A,
268 .period = TIMER_PERIOD_US,
269 .func_p = TemporizadorMedicion,
270 .param_p = NULL
271 };
272 TimerInit(&timer_medir);
273 TimerStart(timer_medir.timer);
274
275 // La tarea principal puede terminar
276}
