101static void vTaskSensorMQ3(void *pvParameters) {
102 (void)pvParameters;
103 char mensaje_bluetooth[32];
104 uint32_t acumulador_adc = 0;
105 uint32_t cantidad_muestras = 0;
106
107 for (;;) {
108 uint16_t lectura_adc = 0;
109 TickType_t ahora = xTaskGetTickCount();
110
111 if (booleano_medir) {
112 if ((int32_t)(medicion_fin_tick - ahora) > 0) {
113 // Al iniciar un nuevo ciclo, vaciar el acumulador de ráfaga
114 if (cantidad_muestras == 0) acumulador_adc = 0;
115
116 AnalogInputReadSingle(CH1, &lectura_adc);
117 acumulador_adc += lectura_adc;
118 cantidad_muestras++;
119 } else {
120 booleano_medir = false;
121 BleSendString("*E0*"); // Apagar el indicador de soplado en la App
122 vTaskDelay(50 / portTICK_PERIOD_MS);
123
124 //A partir de acá se realizan los calculos necesarios para obtener el tiempo de espera
125 float promedio_adc = (float)acumulador_adc / (float)cantidad_muestras;
126 cantidad_muestras = 0;
127
128 // Paso de mV a V y multiplico x2 por el divisor de tension
129 float voltaje_sensor = (promedio_adc / 1000.0f) * 2.0f;
130 // Resistencia en aire limpio (R_o) del MQ-3, valor típico de fábrica
131 float Ro = 5463.0f;
132 float alcohol_sangre_gl = 0.0000f; // Inicializamos por defecto en cero absoluto
133
134 // Si hay suficiente voltaje, significa que hay alcohol y calculamos las curvas exponenciales
135 if (voltaje_sensor > 0.2f) {
136 // 1. Calcular la resistencia dinámica del gas del MQ-3 (R_L = 1000 Ohm)
137 float Rs = 1000.0f * ((5.0f - voltaje_sensor) / voltaje_sensor);
138
139 // 2. Concentración en Aire (mg/L) mediante la regresión exponencial (R_o = 5463 Ohm)
140 float alcohol_aire_mgl = 0.4091f * powf((Rs / Ro), -1.497f);
141
142 // 3. Conversión de unidades aire-sangre mediante el factor clínico 2.0
143 float calculo_sangre = alcohol_aire_mgl * 2.0f;
144
145 // Aplicamos el umbral mínimo de zona muerta para limpiar ruidos residuales
146 if (calculo_sangre >= UMBRAL_ALCOHOL_MINIMO) {
147 alcohol_sangre_gl = calculo_sangre; // Solo se actualiza si supera el mínimo clínico
148 }
149 }
150 else {
151 // ELSE: Si el voltaje es <= 0.2V, es aire limpio absoluto.
152 // No calculamos Rs (evitamos dividir por cero o desbordar powf) y se mantiene en 0.0000 g/L
153 alcohol_sangre_gl = 0.0000f;
154 }
155
156 // 4. Ecuación lineal de depuración hepática (Tasa de Widmark)
157 float beta = 0.15f;
158 float horas_espera = alcohol_sangre_gl / beta;
159
160 //Se envían los resultados a la aplicacion en el celular con sus respectivas etiquetas
161
162 //Tiempo de espera en minutos (Identificador 'T')
163 snprintf(mensaje_bluetooth, sizeof(mensaje_bluetooth), "*T%d*", horas_espera);
164 BleSendString(mensaje_bluetooth);
165 vTaskDelay(50 / portTICK_PERIOD_MS);
166
167 //Concentración directa en sangre (Identificador 'S')
168 snprintf(mensaje_bluetooth, sizeof(mensaje_bluetooth), "*S%.2f*", alcohol_sangre_gl);
169 BleSendString(mensaje_bluetooth);
170 vTaskDelay(50 / portTICK_PERIOD_MS);
171
172 //Voltaje Promedio Real para control de laboratorio (Identificador 'V')
173 snprintf(mensaje_bluetooth, sizeof(mensaje_bluetooth), "*V%.2f*", voltaje_sensor);
174 BleSendString(mensaje_bluetooth);
175 }
176 }
177
178 vTaskDelay(INTERVALO_MUESTREO_MS / portTICK_PERIOD_MS);
179 }
